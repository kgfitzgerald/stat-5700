\subsection{\texorpdfstring{\textbf{Problem
2}}{Problem 2}}\label{problem-2}

Suppose \(P(A) = 0.45\), \(P(B) = 0.30\), and \(P(A \cup B) = 0.6\).

\subsubsection{\texorpdfstring{a. Find
\(P(A \cap B)\)}{a. Find P(A \textbackslash cap B)}}\label{a.-find-pa-cap-b}

\[
\textcolor{red}{P(A \cap B) = P(A) + P(B) - P(A \cup B) = 0.45 + 0.30 - 0.60 = 0.15}
\]

\subsubsection{\texorpdfstring{b. Find
\(P(A' \cap B')\)}{b. Find P(A\textquotesingle{} \textbackslash cap B\textquotesingle)}}\label{b.-find-pa-cap-b}

\[
\textcolor{red}{P(A' \cap B') = P[(A \cup B)']=1 - P(A \cup B) = 1 - 0.60 = 0.40}
\]

\subsection{\texorpdfstring{\textbf{Problem
3}}{Problem 3}}\label{problem-3}

Suppose you draw one card at random from a standard 52-card deck.

Events: - \(A\): The card is red\\
- \(B\): The card is black\\
- \(C\): The card is a heart

For each pair of events, indicate whether they are mutually exclusive,
mutually exhaustive, both, or neither.

\subsubsection{\texorpdfstring{a. \(A\) and
\(B\)}{a. A and B}}\label{a.-a-and-b}

\textcolor{red}{Mutually exclusive and mutually exhaustive}

\subsubsection{\texorpdfstring{b. \(A\) and
\(C\)}{b. A and C}}\label{b.-a-and-c}

\textcolor{red}{Neither}

\subsubsection{\texorpdfstring{c.~\(B\) and
\(C\)}{c.~B and C}}\label{c.-b-and-c}

\textcolor{red}{Mutually exclusive}
